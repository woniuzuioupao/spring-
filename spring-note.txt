class D{
    String name;
    //long,Date,string都行
    //若为string类型，只能是两种格式
    //1."2012-12-12" 
    //2."2012-12-12 12:12:12"
    int time;


    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getTime() {
        return time;
    }

    public void setTime(int time) {
        this.time = time;
    }
}
class E{
    String name;
    //格式多样
    //1.format = "yyyy-MM-dd"
    //2.format = "yyyy-MM"
    //3.format = "yyyy-MM-dd hh:mm:ss"
    //4.format = "yyyy-MM-dd hh:mm"
    //......
    @JSONField(format = "yyyy-MM-dd hh:mm")
    Date time;

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Date getTime() {
        return time;
    }

    public void setTime(Date time) {
        this.time = time;
    }

    @Override
    public String toString() {
        return "E{" +
                "name='" + name + '\'' +
                ", time=" + time +
                '}';
    }
}
